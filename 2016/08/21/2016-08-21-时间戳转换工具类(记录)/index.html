时间戳转换工具类

Posted on 2016-08-21   |  

又是很久没写博客，说好了的一周一更呢？自己又偷懒了5555，今天一下子交了三个月的房租，小金库被掏空了，又有写代码的动力了。哎，犯贱啊..记录下前两天写的一个工具类，以后就不用每次再写了，用到的时候直接拿来用就是了。话不多说，贴代码√

public class TimeUtils {

    public static final SimpleDateFormat DEFAULT_SDF = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss", Locale.getDefault());


    public TimeUtils() {
    }

    /**
     * 时间戳转换字符串
     * <p>Millisecond 毫秒</p>
     * @param millisecond
     * @return
     */
    public static String millisecondToString(long millisecond) {
        return millisecondToString(millisecond, DEFAULT_SDF);
    }


    /**
     * 时间戳转换字符串
     * <p>时间类型自定义</p>
     */

    public static String millisecondToString(long millisecond, SimpleDateFormat sdf) {
        return sdf.format(new Date(millisecond));
    }


    /**
     * 字符串转时间戳(毫秒)
     * <p>格式为yyyy-MM-dd HH:mm:ss</p>
     */
    public static long stringToMillisecond(String time) {
        return stringToMillisecond(time, DEFAULT_SDF);
    }


    /**
     * 字符串转时间戳
     * <p>自定义时间类型</p>
     *
     * @param time 时间字符串
     * @return
     */
    public static long stringToMillisecond(String time, SimpleDateFormat sdf) {
        try {
            return sdf.parse(time).getTime();
        } catch (ParseException e) {
            e.printStackTrace();
        }
        return -1;
    }

    /**
     * 时间字符串转Date(毫秒)
     *
     * @param time
     * @return
     */
    public static Date stringToDate(String time) {
        return stringToDate(time, DEFAULT_SDF);
    }

    /**
     * 时间字符串转Date
     * <p>自定义时间类型</p>
     *
     * @param time
     * @param sdf
     * @return
     */
    public static Date stringToDate(String time, SimpleDateFormat sdf) {
        return new Date(stringToMillisecond(time, sdf));
    }


    /**
     * Date时间转时间字符串(毫秒)
     *
     * @param datetime 时间字符串
     * @return
     */
    public static String dateToString(Date datetime) {
        return dateToString(datetime, DEFAULT_SDF);
    }

    /**
     * Date时间转字符串
     * <p>自定义时间类型</p>
     *
     * @param datetime date时间
     * @param sdf      时间类型
     * @return
     */
    private static String dateToString(Date datetime, SimpleDateFormat sdf) {
        return sdf.format(datetime);
    }

    /**
     * Date转时间戳
     * @param datetime  Date时间
     * @return
     */
    public static long dateToMillisecond(Date datetime) {
        return datetime.getTime();
    }

    /**
     * 时间戳转Date
     * @param millisecond  时间戳
     * @return
     */
    public static Date millisecondToDate(long millisecond) {
        return new Date(millisecond);
    }

    /**
     *  获得系统当前时间戳
     *  <p>系统默认毫秒yyyy-MM-dd HH:mm:ss</p>
     * @return
     */
    public static long getNowMillisecond() {
        return System.currentTimeMillis();
    }

}
